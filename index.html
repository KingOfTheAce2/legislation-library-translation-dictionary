Quick Start

// JavaScript Example const API_URL = 'https://kingoftheace2.github.io/legislation-library-translation-dictionary/Glossary-of-Dutch-Procedural-Terminology.json'; async function searchTerm(query) { const response = await fetch(API_URL); const terms = await response.json(); return terms.filter(term => term.source.includes(query) || term.target.includes(query) ); } // Usage const results = await searchTerm('aanhangig'); console.log(results);

Python Example

import requests API_URL = 'https://kingoftheace2.github.io/legislation-library-translation-dictionary/Glossary-of-Dutch-Procedural-Terminology.json' response = requests.get(API_URL) terms = response.json() # Search for a term results = [t for t in terms if 'pending' in t['target']] print(f"Found {len(results)} matches")
